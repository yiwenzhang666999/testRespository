treeHelper.bindEvent

概述

绑定树事件。

关于树的事件监听的入口。

Function 参数说明

eventTargetstring事件对象(treeHelper.eventTarget.BHDK,treeHelper.eventTarget.YWSJ)

eventTypestring事件类型(treeHelper.events.ADD_CLICK,MORE_CLICK,DEL_CLICK,CREATE_CLICK,CHECKED)

callbackfunction回调函数

function 举例

var tree = treeHelper.getTree("zyjg",{expandLevel : 1},[{id:1,text:"test",children[]}]);
tree.bindEvent(treeHelper.eventTarget.BHDK,treeHelper.events.MORE_CLICK,function(node){
	console.log(node);
});
tree.bindEvent(treeHelper.eventTarget.YWSJ,treeHelper.events.ADD_CLICK,function(node){
	alert("add---->"+node.id);
});
tree.bindEvent(treeHelper.eventTarget.YWSJ,treeHelper.events.CREATE_CLICK,function(node){
	alert("create---->"+node.id);
});
tree.bindEvent(treeHelper.eventTarget.BHDK,treeHelper.events.DEL_CLICK,function(node){
	alert("del---->"+node.id);
});
